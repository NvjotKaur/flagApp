import React, { Component } from 'react';
import SearchBox from '../searchBoxComponent/searchBoxComponent';
import './countryFlagPicker.css';
 
var continents = require('../../data/continents.json');
 
class countryFlagPicker extends Component {
    constructor(props) {
        super(props);
 
        this.state = {
            continents: continents,
            continentList: [],
            countriesList: [],
            selectedContinent: null,
            selectedCountries: [],
        }
        this.onContinentSelection = this.onContinentSelection.bind(this);
        this.onContrySelection = this.onContrySelection.bind(this);
        this.clearFlags= this.clearFlags.bind(this);
    }
 
    componentDidMount() {
        this.configureContentList();
    }
 
    configureContentList() {
        const { continents } = this.state;
        const items = continents.map((item) => {
            return {
                label: item.continent,
                value: item.continent
            }
        });
        this.setState({ continentList: items })
    }
 
    onContinentSelection(selectedContinent, shouldClear) {
        if(shouldClear){
            this.setState({
                countriesList: [],
                selectedContinent: null,
                selectedCountries:[]
            });
            return;
        }
        const { continents } = this.state;
        const currentContries1 = continents.find(item => item.continent === selectedContinent.value).countries;
        const currentContries = currentContries1.map((item) => {
            return {
                label: item.name,
                value: item.flag
            }
        })
        this.setState({
            countriesList: currentContries,
            selectedContinent: selectedContinent,
            selectedCountries:[]
        });
    }
 
    onContrySelection(value, shouldClear) {
        if(shouldClear){
            this.setState({ selectedCountries: [] });
            return;
        }
        this.setState({ selectedCountries: value });
    }
 
    clearFlags(){
        this.setState({ selectedCountries: [] });
    }
 
    render() {
        const { continentList, countriesList, selectedContinent, selectedCountries } = this.state;
        return (
            <div className="home-container">
                <div className="home-flag">
                    <h1>Flag Picker</h1>
                    <p>This app will help you to learn flags around the world in 3 steps</p>
                </div>
                <div className="content">
                    <div className="home-content">
                        {continentList.length ? <div className="continent-picker-container">
                            <h3>Step 1</h3>
                            <SearchBox label="select a continent" options={continentList} multiSelect={false} onOptionSelection={this.onContinentSelection} />
                        </div> : ''}
                        {selectedContinent ?
                            <div className="selected-country"><p>You Selected</p><h3>{selectedContinent.label}</h3></div> : ''}
                    </div>
                    <div className="home-content">
                        {countriesList.length ? <div className="country-selection-container">
                            <h3>Step 2</h3>
                            <SearchBox label="select the Countries" options={countriesList} multiSelect={true} onOptionSelection={this.onContrySelection} />
                        </div> : ''}
                    </div>
                    <div className="home-content">
                        {selectedCountries.length ? <div className="home-content-flag">
                            <h3>Selected flags:</h3>
                            <div>
                                {selectedCountries.map((country, index) => {
                                    return (<img
                                        key={index}
                                        alt={country.label}
                                        width="100"
                                        height="100"
                                        src={`http://www.countryflags.io/${country.value}/shiny/64.png`} />)
                                })}
                            </div>
                            <button onClick={this.clearFlags}>Clear flags</button>
                        </div> : ''}
                    </div>
 
                </div>
            </div>
        );
    }
}
 
export default countryFlagPicker;
